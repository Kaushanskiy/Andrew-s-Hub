#include <stdio.h> //Подключение библиотеки ввода/вывода  
#include <string.h>  
#include <math.h> //Подключение библиотеки математических функций  
#include <locale.h>  
//Точка входа в программу - главная функция  
int main() {
    setlocale(LC_ALL, "Russian");
    int x1, x2, y1, y2, x3, y3;
    float a, b, c, S, P;

    //Вычисляем стороны между тремя точками с 
    // координатами (x1,y1) и (x2,y2) и (x3,y3) 
    printf("Три точки с координатами (x1,y1) и (x2,y2) и (x3,y3)\n");
    printf("Введите координаты первой точки: ");
    scanf_s("%d" "%d", &x1, &y1);
    printf("Введите координаты второй точки: ");
    scanf_s("%d" "%d", &x2, &y2);
    printf("Введите координаты третьей точки: ");
    scanf_s("%d" "%d", &x3, &y3);

    //Рассчитываем площадь и периметр треугольника
    S = 0.5 * abs(((x1 - x3) * (y2 - y3) - (x2 - x3) * (y1 - y3)));
    a = sqrt(pow((x2 - x1), 2) + pow((y2 - y1), 2));
    b = sqrt(pow((x3 - x2), 2) + pow((y3 - y2), 2));
    c = sqrt(pow((x3 - x1), 2) + pow((y3 - y1), 2));
    printf("Сторона прямоугольника a %6.2f\n", a);
    printf("Сторона прямоугольника b %6.2f\n", b);
    printf("Сторона прямоугольника с %6.2f\n", c);
    P = (a + b + c);

    //Вывод результатов  
    printf("Площадь прямоугольника %7.3f\n", S);
    printf("Периметр прямоугольника %7.3f\n", P);
    return 0;

}
