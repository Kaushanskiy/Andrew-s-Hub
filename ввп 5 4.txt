#include <stdio.h> //Подключение библиотеки ввода/вывода  
#include <string.h>  
#include <math.h> //Подключение библиотеки математических функций  
#include <locale.h>  
//Точка входа в программу - главная функция  
int main() {
    setlocale(LC_ALL, "Russian");
    int x1, x2, y1, y2;
    float a, b, S, P;

    //Вычисляем стороны между двумя точками с координатами (x1,y1) и (x2,y2) 
    printf("Две точки с координатами (x1,y1) и (x2,y2)\n");
    printf("Введите координаты первой точки: ");
    scanf_s("%d" "%d", &x1, &y1);
    printf("Введите координаты второй точки: ");
    scanf_s("%d" "%d", &x2, &y2);
    a = sqrt(pow((x1 - x1), 2) + pow((y2 - y1), 2));
    b = sqrt(pow((x2 - x1), 2) + pow((y2 - y2), 2));
    printf("Сторона прямоугольника a %6.2f\n", a);
    printf("Сторона прямоугольника b %6.2f\n", b);
    //Рассчитываем площадь и периметр прямоугольника
    S = a * b;
    P = 2 * (a + b);

    //Вывод результатов  
    printf("Площадь прямоугольника %7.3f\n", S);
    printf("Периметр прямоугольника %7.3f\n", P);
    return 0;

}
